import javafx.application.Application;
import javafx.fxml.FXMLLoader;
import javafx.scene.Parent;
import javafx.scene.Scene;
import javafx.stage.Stage;

public class MainApp extends Application {
    
    private static Stage stagePrincipal; // Guarda a janela para permitir a troca de telas

    @Override
    public void start(Stage primaryStage) throws Exception {
        stagePrincipal = primaryStage;
        
        // Configura para abrir PRIMEIRO na tela de cadastro de veículos
        // Lembre-se: O arquivo "GestaoFrotas.fxml" deve estar dentro da pasta "view" no seu projeto
        trocarTela("/view/GestaoFrotas.fxml", "Passo 1: Cadastro de Veículos");
    }

    /**
     * Método utilitário para mudar de tela dinamicamente durante a execução.
     * Como é um método static, usamos 'MainApp.class.getResource' para localizar os arquivos.
     */
    public static void trocarTela(String caminhoFxml, String titulo) {
        try {
            // 1. Carrega o arquivo FXML usando o caminho informado
            FXMLLoader loader = new FXMLLoader(MainApp.class.getResource(caminhoFxml));
            Parent root = loader.load(); 
            
            // 2. Configura e exibe a nova cena na janela principal
            stagePrincipal.setTitle(titulo);
            stagePrincipal.setScene(new Scene(root, 1000, 600));
            stagePrincipal.show();
            
        } catch (Exception e) {
            System.err.println("Erro ao carregar a tela: " + caminhoFxml);
            e.printStackTrace();
        }
    }

    public static void main(String[] args) {
        // Inicia a aplicação JavaFX
        launch(args);
    }
}

import javafx.collections.FXCollections;

public class CadastroSensoresController {

    @FXML private ComboBox<Veiculo> comboVeiculoSensor;
    @FXML private TextField txtNomeSensor;
    @FXML private ComboBox<String> comboTipoSensor;
    @FXML private Button btnInstalarSensor;
    @FXML private TableView<Sensor> tblSensores;
    @FXML private TableColumn<Sensor, String> colSensorNome;
    @FXML private TableColumn<Sensor, String> colSensorTipo;

    private static int contadorSensorId = 100;

    @FXML
    public void initialize() {
        // Preenche o ComboBox com os veículos criados na primeira tela
        comboVeiculoSensor.setItems(GestaoFrotaController.getListaVeiculosGlobal());
        
        // Define os tipos de sensores baseados nas classes que você possui no modelo
        comboTipoSensor.getItems().addAll("Velocidade", "Temperatura", "Lógico (On/Off)", "Geográfico (GPS)");

        // Configura as colunas da tabela de visualização
        colSensorNome.setCellValueFactory(new PropertyValueFactory<>("nome"));
        colSensorTipo.setCellValueFactory(new PropertyValueFactory<>("categoria"));
        
        // Atualiza a tabela dinamicamente ao trocar de veículo
        comboVeiculoSensor.getSelectionModel().selectedItemProperty().addListener((obs, antigo, novoVeiculo) -> {
            if (novoVeiculo != null) {
                // Usa o método correto: getConfiguracao() que retorna a List<Sensor>
                tblSensores.setItems(FXCollections.observableArrayList(novoVeiculo.getConfiguracao()));
            } else {
                tblSensores.getItems().clear();
            }
        });
    }

    @FXML
    private void handleInstalarSensor() {
        Veiculo veiculoSelecionado = comboVeiculoSensor.getValue();
        String nome = txtNomeSensor.getText();
        String tipo = comboTipoSensor.getValue();

        if (veiculoSelecionado == null || nome == null || nome.trim().isEmpty() || tipo == null) {
            System.out.println("Erro: Selecione o veículo, dê um nome e escolha o tipo do sensor.");
            return;
        }

        Sensor novoSensor;
        int id = contadorSensorId++;

        // Instanciação polimórfica respeitando estritamente os construtores das suas classes filhas
        switch (tipo) {
            case "Velocidade":
                novoSensor = new SensorVelocidade(id, nome.trim(), "Telemetria", 0.0);
                break;
            case "Temperatura":
                novoSensor = new SensorTemperatura(id, nome.trim(), "Telemetria", 20.0);
                break;
            case "Lógico (On/Off)":
                novoSensor = new SensorLogico(id, nome.trim(), "Segurança", false);
                break;
            case "Geográfico (GPS)":
                novoSensor = new SensorGeografico(id, nome.trim(), 100.0, 100.0);
                break;
            default:
                return;
        }

        // Associa o dispositivo ao carro usando o método adicionarSensor do seu modelo
        veiculoSelecionado.adicionarSensor(novoSensor);

        // Atualiza a lista da tabela
        tblSensores.setItems(FXCollections.observableArrayList(veiculoSelecionado.getConfiguracao()));
        
        txtNomeSensor.clear();
        System.out.println("Sensor '" + nome + "' instalado no veículo " + veiculoSelecionado.getIdentificador());
    }
}

    /**
     * Aplica a variação física simulada convertendo o dado para a sua classe especializada correspondente
     */
    private void alterarMecanismoSensor(int indice, double variacao) {
        Sensor sensor = tblSensoresSimulacao.getSelectionModel().getSelectedItem();
        if (sensor == null) return;

        double[] valoresAtuais = sensor.getValores();

        // Faz o tratamento com base em cada uma das subclasses de sensores do seu modelo
        if (sensor instanceof SensorVelocidade) {
            SensorVelocidade sv = (SensorVelocidade) sensor;
            sv.setValorAtual(valoresAtuais[0] + variacao);
        } 
        else if (sensor instanceof SensorTemperatura) {
            SensorTemperatura st = (SensorTemperatura) sensor;
            st.setValorAtual(valoresAtuais[0] + variacao);
        } 
        else if (sensor instanceof SensorLogico) {
            SensorLogico sl = (SensorLogico) sensor;
            // Alterna o estado booleano baseado no sinal do clique
            sl.setValorAtual(valoresAtuais[0] + variacao > 0.5);
        } 
        else if (sensor instanceof SensorGeografico) {
            SensorGeografico sg = (SensorGeografico) sensor;
            double novaLat = valoresAtuais[0] + (indice == 0 ? variacao : 0);
            double novaLong = valoresAtuais[1] + (indice == 1 ? variacao : 0);
            // Sobrescreve instanciando as novas posições simuladas do GPS
            int idAtual = tblSensoresSimulacao.getSelectionModel().getSelectedIndex();
            Veiculo v = comboVeiculoMonitoramento.getValue();
            if (v != null) {
                // Atualiza o estado recriando o sensor geográfico na lista
                SensorGeografico novoGps = new SensorGeografico(1, sg.getNome(), novaLat, novaLong);
                v.getConfiguracao().set(v.getConfiguracao().indexOf(sg), novoGps);
                tblSensoresSimulacao.getSelectionModel().select(novoGps);
            }
        }

        // Atualiza a interface gráfica da tabela
        tblSensoresSimulacao.refresh();

        // Dispara o motor lógico para recalcular os circuitos AND combinatórios
        gerenciadorAlerta.processarMonitoramento();

        // Atualiza os alarmes vigentes na tela
        atualizarPainelDinamico();
    }

    private void atualizarPainelDinamico() {
        Veiculo veiculoAtual = comboVeiculoMonitoramento.getValue();
        if (veiculoAtual == null) return;

        // 1. Atualiza a lista visual de alertas críticos correntes (Filtra pelo veículo ativo)
        listAlertasAtivos.getItems().clear();
        for (RegistroAlerta registro : gerenciadorAlerta.obterAlertasAtivos()) {
            if (registro.getCondicao().getVeiculo() != null && 
                registro.getCondicao().getVeiculo().getIdentificador().equals(veiculoAtual.getIdentificador())) {
                
                listAlertasAtivos.getItems().add(
                    "🚨 [" + registro.getCondicao().getNomeRegra() + "] -> " + registro.getCondicao().getMensagemAlerta()
                );
            }
        }

        // 2. Atualiza a tabela histórica (Log global completo de auditoria)
        tblHistorico.setItems(FXCollections.observableArrayList(gerenciadorAlerta.getHistoricoCompleto()));
    }

    public static GerenciadorAlerta getGerenciadorAlerta() {
        return gerenciadorAlerta;
    }
}

package model;

import java.util.ArrayList;
import java.util.List;

/**
 * Representa a regra de negócio do alerta combinatório (Porta AND).
 * Agrupa todos os gatilhos necessários que devem ser simultaneamente verdadeiros
 * para disparar um alerta para um veículo específico.
 */
public class CondicaoAlerta {
    
    private String nomeRegra;
    private String mensagemAlerta;
    protected Veiculo veiculo;
    
    // Lista única tipada com a nossa estrutura de dados GatilhoSensor
    protected List<GatilhoSensor> sensoresMonitorados = new ArrayList<>();
    protected List<Boolean> microCondicao = new ArrayList<>();
    
    public CondicaoAlerta(String nomeRegra, String mensagem) {
        this.nomeRegra = nomeRegra;
        this.mensagemAlerta = mensagem; 
    }
    
    public void adicionarVeiculo(Veiculo v) { 
        this.veiculo = v;
    }
    
    /**
     * Sobrecarga 1: Para sensores comuns de valor único (Velocidade, Temperatura, Porta, etc.).
     * Automaticamente define o índice de leitura do array como 0.
     */
    public void adicionarSensores(Sensor s, Comparador comparador, double valorRef) {
        GatilhoSensor novoGatilho = new GatilhoSensor(s, 0, comparador, valorRef);
        this.sensoresMonitorados.add(novoGatilho);
    }

    /**
     * Sobrecarga 2: Específica para o SensorGeografico (GPS) ou outros sensores multi-valor.
     * Permite especificar qual posição do array testar (0 para Latitude, 1 para Longitude).
     */
    public void adicionarSensores(Sensor s, int indice, Comparador comparador, double valorRef) {
        GatilhoSensor novoGatilho = new GatilhoSensor(s, indice, comparador, valorRef);
        this.sensoresMonitorados.add(novoGatilho);
    }
    
    /**
     * Motor do circuito lógico combinatório. 
     * Varre todos os gatilhos, busca os valores reais atualizados e faz a soma AND.
     */
    public boolean avaliarCondicaoGeral() {
        if (sensoresMonitorados.isEmpty()) {
            return false; // Se não houver regras configuradas, o circuito retorna falso
        }

        // Limpa os resultados parciais anteriores antes de uma nova checagem em tempo real
        this.microCondicao.clear();
        
        // 1. Fase de Varredura: Calcula o resultado lógico individual de cada gatilho
        for (GatilhoSensor gatilho : sensoresMonitorados) {
            
            // Pega o array double[] definido na classe abstrata Sensor
            double[] valoresDoSensor = gatilho.getSensor().getValores();
            
            // Extrai o valor na posição configurada (0 para comuns, 0 ou 1 para GPS)
            double valorSensor = valoresDoSensor[gatilho.getIndiceValor()];
            
            boolean resultadoMicro = false;
            Comparador operacao = gatilho.getComparador();
            double ref = gatilho.getValorReferencia();
            
            // Avalia o critério com base no seu Enum Comparador
            if (operacao == Comparador.MAIOR_QUE) {
                resultadoMicro = valorSensor > ref;
            } else if (operacao == Comparador.MENOR_QUE) {
                resultadoMicro = valorSensor < ref;
            } else if (operacao == Comparador.IGUAL) {
                resultadoMicro = valorSensor == ref;
            } else if (operacao == Comparador.DIFERENTE) {
                resultadoMicro = valorSensor != ref;
            }
            
            // Adiciona o bit resultante (true/false) na lista de parciais
            this.microCondicao.add(resultadoMicro);
        }
        
        // 2. Fase Combinatória: Multiplicação acumulada AND (&) entre todas as micro-condições
        boolean resultadoAnd = true;
        
        for (boolean bitEntrada : microCondicao) {
            resultadoAnd = resultadoAnd & bitEntrada;
            
            // Se encontrar qualquer bit falso (0), o AND inteiro quebra na hora
            if (!resultadoAnd) {
                return false;
            }
        }
        
        return resultadoAnd;
    }

    // Getters para a camada de visão (JavaFX) ou gerenciador de alertas
    public String getNomeRegra() { return nomeRegra; }
    public String getMensagemAlerta() { return mensagemAlerta; }
    public Veiculo getVeiculo() { return veiculo; }
}

import java.sql.*;
import com.j256.ormlite.jdbc.JdbcConnectionSource;

public class Database
{
   private String databaseName = null;
   private JdbcConnectionSource connection = null;
   
   public Database(String databaseName) {
       this.databaseName = databaseName;
   }    
   
   public JdbcConnectionSource getConnection() throws SQLException {
      if ( databaseName == null ) {
          throw new SQLException("database name is null");
      }
      if ( connection == null ) {
          try {
              connection = new JdbcConnectionSource("jdbc:sqlite:"+databaseName);             
            } catch ( Exception e ) {
                System.err.println( e.getClass().getName() + ": " + e.getMessage() );
                System.exit(0);
            }
      }
      return connection;
   }
   
   public void close() {
       if ( connection != null ) {
           try {
               connection.close();
               this.connection = null;
           } catch (java.lang.Exception e) {
               System.err.println(e);
           }
       }
   }
}

package model;

/**
 * Representa a configuração de um gatilho individual para um sensor.
 * Une o dispositivo físico a um critério de teste específico usando o Enum Comparador.
 */
public class GatilhoSensor {
    
    private Sensor sensor;
    private int indiceValor;
    private Comparador comparador; 
    private double valorReferencia;

    /**
     * Construtor completo para instanciar a regra de teste do sensor.
     */
    public GatilhoSensor(Sensor sensor, int indiceValor, Comparador comparador, double valorReferencia) {
        this.sensor = sensor;
        this.indiceValor = indiceValor;
        this.comparador = comparador;
        this.valorReferencia = valorReferencia;
    }

    // =========================================================================
    // GETTERS
    // =========================================================================
    
    public Sensor getSensor() {
        return this.sensor;
    }

    public int getIndiceValor() {
        return this.indiceValor;
    }

    public Comparador getComparador() {
        return this.comparador;
    }

    public double getValorReferencia() {
        return this.valorReferencia;
    }
}

 */
public class GerenciadorAlerta {
    
    // Lista de circuitos lógicos que o sistema deve vigiar
    private List<CondicaoAlerta> regrasCadastradas = new ArrayList<>();
    
    // O histórico completo de alertas (ativos e encerrados) para o painel
    private List<RegistroAlerta> historicoAlertas = new ArrayList<>();

    public void cadastrarRegra(CondicaoAlerta condicao) {
        this.regrasCadastradas.add(condicao);
    }

    /**
     * Este método roda a cada atualização do sistema (Tick do relógio).
     * Ele avalia os circuitos e gerencia o nascimento e morte dos alertas no histórico.
     */
    public void processarMonitoramento() {
        for (CondicaoAlerta regra : regrasCadastradas) {
            
            // Pergunta para o circuito lógico se a condição de perigo está ativa AGORA
            boolean emPerigoAgora = regra.avaliarCondicaoGeral(); // Nota: usando avaliarCondicaoGeral()
            
            // Procura se já existe um alerta aberto na história para essa mesma regra
            RegistroAlerta alertaAtual = buscarAlertaAtivoPorRegra(regra);

            if (emPerigoAgora) {
                // CENÁRIO A: Está em perigo e não tinha alerta aberto ainda -> Cria um novo!
                if (alertaAtual == null) {
                    RegistroAlerta novoAlerta = new RegistroAlerta(regra);
                    this.historicoAlertas.add(novoAlerta);
                    System.out.println("🚨 [NOVO ALERTA] " + regra.getNomeRegra() + " iniciado para o veículo " + regra.getVeiculo().getIdentificador());
                }
                // Se já existia (alertaAtual != null), não faz nada. O tempo continua correndo.
                
            } else {
                // CENÁRIO B: O veículo voltou ao normal, mas havia um alerta aberto -> Encerra ele!
                if (alertaAtual != null) {
                    alertaAtual.encerrarAlerta();
                    System.out.println("✅ [ALERTA RESOLVIDO] " + regra.getNomeRegra() + " finalizado. Duração: " + alertaAtual.getDuracaoFormatada());
                }
            }
        }
    }

    /**
     * Busca no histórico se existe algum alerta que ainda está aberto (ativo) para uma regra.
     */
    private RegistroAlerta buscarAlertaAtivoPorRegra(CondicaoAlerta regra) {
        for (RegistroAlerta registro : historicoAlertas) {
            if (registro.isAtivo() && registro.getCondicao() == regra) {
                return registro;
            }
        }
        return null; // Não há alerta em aberto para essa regra
    }

    /**
     * Retorna apenas os alertas que estão acontecendo neste exato momento (Para o painel piscar em vermelho)
     */
    public List<RegistroAlerta> obterAlertasAtivos() {
        List<RegistroAlerta> ativos = new ArrayList<>();
        for (RegistroAlerta r : historicoAlertas) {
            if (r.isAtivo()) ativos.add(r);
        }
        return ativos;
    }

    /**
     * Retorna o histórico completo (Para preencher uma tabela de auditoria)
     */
    public List<RegistroAlerta> getHistoricoCompleto() {
        return this.historicoAlertas;
    }
}

package model;

import java.time.LocalDateTime;
import java.time.Duration;

/**
 * Representa um evento de alerta que aconteceu no histórico do sistema.
 * Registra o momento de início, fim e calcula a duração do alerta.
 */
public class RegistroAlerta {
    
    private CondicaoAlerta condicao;
    private LocalDateTime horarioInicio;
    private LocalDateTime horarioFim;
    private boolean ativo;

    public RegistroAlerta(CondicaoAlerta condicao) {
        this.condicao = condicao;
        this.horarioInicio = LocalDateTime.now(); // Captura a hora exata do disparo
        this.ativo = true; // Nasce ativo
    }

    /**
     * Finaliza o alerta quando a situação do veículo volta ao normal.
     */
    public void encerrarAlerta() {
        this.horarioFim = LocalDateTime.now();
        this.ativo = false;
    }

    /**
     * Calcula há quanto tempo o alerta está ativo ou quanto tempo ele durou.
     */
    public String getDuracaoFormatada() {
        LocalDateTime fim = (this.ativo) ? LocalDateTime.now() : this.horarioFim;
        Duration duracao = Duration.between(horarioInicio, fim);
        
        long segundos = duracao.toSeconds();
        if (segundos < 60) {
            return segundos + " segundos";
        }
        return duracao.toMinutes() + " minutos e " + (segundos % 60) + " segundos";
    }

    // =========================================================================
    // GETTERS
    // =========================================================================
    public CondicaoAlerta getCondicao() { return condicao; }
    public LocalDateTime getHorarioInicio() { return horarioInicio; }
    public LocalDateTime getHorarioFim() { return horarioFim; }
    public boolean isAtivo() { return ativo; }
}

        this.entityClass = entityClass;
        setDatabase(database);
        loadedEntities = new ArrayList<>();
    }

    public void setDatabase(Database database) {
        Repositorio.database = database;
        try {
            dao = DaoManager.createDao(database.getConnection(), entityClass);
            TableUtils.createTableIfNotExists(database.getConnection(), entityClass);
        } catch (SQLException e) {
            System.out.println(e);
        }
    }

    public T create(T entity) {
        try {
            int nrows = dao.create(entity);
            if (nrows == 0)
                throw new SQLException("Error: object not saved");
            this.loadedEntity = entity;
            loadedEntities.add(entity);
        } catch (SQLException e) {
            System.out.println(e);
        }
        return entity;
    }

    public void update(T entity) {
        try {
            dao.update(entity);
        } catch (SQLException e) {
            System.out.println(e);
        }
    }

    public void delete(T entity) {
        try {
            dao.delete(entity);
        } catch (SQLException e) {
            System.out.println(e);
        }
    }

    public T loadFromId(ID id) {
        try {
            this.loadedEntity = dao.queryForId(id);
            if (this.loadedEntity != null)
                this.loadedEntities.add(this.loadedEntity);
        } catch (SQLException e) {
            System.out.println(e);
        }
        return this.loadedEntity;
    }

        try {
            this.loadedEntities = dao.queryForAll();
            if (!this.loadedEntities.isEmpty())
                this.loadedEntity = this.loadedEntities.get(0);
        } catch (SQLException e) {
            System.out.println(e);
        }
        return this.loadedEntities;
    }

}

package model;
import java.util.Random;

public class SensorGeografico extends Sensor {
    private double latitude;
    private double longitude;
    private static final Random random = new Random();
    
    public SensorGeografico(int id, String nome){
        super(id, nome, "Telemetria");
        this.latitude = 100;
        this.longitude = 100;
        //-16.67744721562849, -49.24250895082047
    }
    
    public void simularDeslocamentoAleatorio() {
        // 1. Sorteia uma distância de 0 a 200 metros
        double distancia = random.nextDouble() * 200.0;
        
        // 2. Sorteia uma direção (ângulo de 0 a 360 graus em radianos)
        double angulo = random.nextDouble() * 2 * Math.PI;

        // 3. Decompõe a distância percorrida nos eixos X e Y
        double deltaX = distancia * Math.cos(angulo);
        double deltaY = distancia * Math.sin(angulo);

        // 4. Aplica o movimento diretamente nas coordenadas atuais
        this.latitude += deltaX;
        this.longitude += deltaY;
    }
    
    
    public SensorGeografico(int id, String nome, double latitude, double longitude) {
        super(id, nome, "Telemetria");
        this.latitude = latitude;
        this.longitude = longitude;
    }

    @Override
    public double[] getValores() {
        return new double[] { this.latitude, this.longitude }; // Array de tamanho 2!
    }
    
    public String getLatLong(){return "latlong = " + latitude + ", " + longitude;}
    
    public void atualizarCoordenadas(double latitude, double longitude) {
        this.latitude = latitude;
        this.longitude = longitude;
    }
}
